from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import Union
import docx
import io
import json
import base64
import os
import re
import google.generativeai as genai

# --- CONFIGURACIÓN DE SEGURIDAD ---
llave_secreta = os.getenv("GEMINI_API_KEY")

if not llave_secreta:
    raise ValueError("No se encontró GEMINI_API_KEY en las variables de entorno.")

genai.configure(api_key=llave_secreta)

modelo = genai.GenerativeModel('gemini-2.5-flash')

app = FastAPI(title="Auditor Legal IA - Enterprise Final")

# --- TIMBRE PARA UPTIMEROBOT ---

async def despertar_servidor():
    return {"estado": "El servidor está despierto y listo."}

@app.head("/")
async def despertar_servidor_head():
    return {"estado": "OK"}

# --- MODELOS DE DATOS OPTIMIZADOS (FIX 422) ---
class PeticionContrato(BaseModel):
    nombre_archivo: str
    # Acepta tanto un string directo como el diccionario anidado de Power Automate
    archivo_base64: Union[str, dict]

class PeticionPregunta(BaseModel):
    archivo_base64: Union[str, dict]
    pregunta: str

# --- ENDPOINT 1: LA AUDITORÍA ESTRICTA (FIX 429 - DIETA DE TOKENS) ---
@app.post("/auditar-contrato")
async def auditar_contrato(peticion: PeticionContrato):
    try:
        texto_b64 = peticion.archivo_base64
        
        # 1. Filtro a prueba de balas para limpiar la basura de Power Automate
        if isinstance(texto_b64, dict):
            texto_b64 = texto_b64.get("contentBytes", "")
        elif isinstance(texto_b64, str) and "contentBytes" in texto_b64:
            try:
                obj = json.loads(texto_b64)
                texto_b64 = obj.get("contentBytes", texto_b64)
            except:
                pass
        
        # 2. Descompresión del Word en memoria
        contenido_binario = base64.b64decode(texto_b64)
        documento_io = io.BytesIO(contenido_binario)
        doc = docx.Document(documento_io)
        texto_completo = "\n".join([parrafo.text for parrafo in doc.paragraphs if parrafo.text.strip()])
        
        # 3. PROMPT MAESTRO OPTIMIZADO (Menos tokens = Más velocidad = No Error 429)
        prompt_maestro = f"""
        Eres un auditor legal experto y MUY ESTRICTO. Evalúa el siguiente contrato basándote ÚNICAMENTE en estas reglas innegociables:

        REGLAS RGPD (5 Controles):
        1. Acuerdo de Tratamiento de Datos (DPA): Exigido adjunto o incluido explícitamente.
        2. Transferencias Internacionales: Prohibidas sin garantías de la UE.
        3. Seguridad y Brechas: Notificación de incidentes en plazo máximo de 24-72h.
        4. Asistencia en Derechos ARCO/Auditorías: Proveedor debe ayudar activamente al cliente.
        5. Destrucción de datos: Eliminación o devolución certificada al finalizar.

        REGLAS DORA (6 Controles):
        1. Descripción de funciones/ubicación: Claridad en qué hace y dónde procesa datos.
        2. SLA y Gestión de incidentes: Tiempos de respuesta claros y penalizaciones.
        3. Derechos de acceso y auditoría sin restricciones: Auditorías permitidas sin preavisos abusivos.
        4. Subcontratación: Prohibida sin autorización o notificación previa.
        5. Estrategias de salida (Exit Plan): Plan de transición ordenado sin costos abusivos.
        6. Medidas de seguridad TIC: Mención a estándares de ciberseguridad.
        
        [CONTRATO A ANALIZAR]
        {texto_completo}
        
        TAREAS OBLIGATORIAS:
        1. Extrae la información básica: proveedor, tipo_contrato, duracion, fecha.
        2. Evalúa EXHAUSTIVAMENTE los 5 controles RGPD y 6 controles DORA mencionados arriba.
        
        REGLAS DE FORMATO ESTRICTAS:
        - 'estado': DEBE SER EXACTAMENTE "OK", "Falta", o "Riesgo".
        - 'observacion': Máximo 12 palabras.
        - 'evidencia_textual': Inicia indicando sección entre corchetes, luego máximo 15 palabras.

        Devuelve ÚNICAMENTE un JSON con esta estructura exacta:
        {{
          "informacion_basica": {{ "proveedor": "", "tipo_contrato": "", "duracion": "", "fecha": "" }},
          "cumplimiento_rgpd": [ {{"control": "", "estado": "", "observacion": "", "evidencia_textual": ""}} ],
          "cumplimiento_dora": [ {{"control": "", "estado": "", "observacion": "", "evidencia_textual": ""}} ]
        }}
        """

        # 4. Invocación a Gemini Forzando JSON puro
        respuesta = await modelo.generate_content_async(
            prompt_maestro,
            generation_config={"response_mime_type": "application/json"}
        )
        datos = json.loads(respuesta.text)

        # 5. Lógica Matemática del Nivel de Riesgo
        lista_estados_rgpd = [item.get("estado", "OK") for item in datos.get("cumplimiento_rgpd", [])]
        lista_estados_dora = [item.get("estado", "OK") for item in datos.get("cumplimiento_dora", [])]
        todos_los_estados = lista_estados_rgpd + lista_estados_dora

        total_faltas = todos_los_estados.count("Falta")
        total_riesgos = todos_los_estados.count("Riesgo")

        if total_faltas >= 2 or total_riesgos >= 3:
            nivel = "Bajo"
            recomendacion = "Revisión legal obligatoria"
        elif total_faltas > 0 or total_riesgos > 0:
            nivel = "Medio"
            recomendacion = "Aprobable con cambios"
        else:
            nivel = "Alto"
            recomendacion = "Aprobable"

        datos["resultado_final"] = {
            "nivel_cumplimiento": nivel,
            "recomendacion": recomendacion
        }
        
        # 🔥 Limpieza de RAM
        del texto_completo
        del prompt_maestro
        
        return datos

    except Exception as e:
        print(f"🔥 ERROR INTERNO EN AUDITORIA: {str(e)}")
        raise HTTPException(status_code=500, detail=f"Error interno: {str(e)}")


# --- ENDPOINT 2: EL NUEVO CHATBOT CONVERSACIONAL ---
@app.post("/preguntar-contrato")
async def preguntar_contrato(peticion: PeticionPregunta):
    try:
        texto_b64 = peticion.archivo_base64
        
        # Limpieza de Base64 homologada
        if isinstance(texto_b64, dict):
            texto_b64 = texto_b64.get("contentBytes", "")
        elif isinstance(texto_b64, str) and "contentBytes" in texto_b64:
            try:
                obj = json.loads(texto_b64)
                texto_b64 = obj.get("contentBytes", texto_b64)
            except:
                pass
        
        contenido_binario = base64.b64decode(texto_b64)
        documento_io = io.BytesIO(contenido_binario)
        doc = docx.Document(documento_io)
        texto_completo = "\n".join([parrafo.text for parrafo in doc.paragraphs if parrafo.text.strip()])
        
        prompt_qa = f"""
        Eres un asistente legal experto. Responde a la pregunta basándote EXCLUSIVAMENTE en el contrato.
        
        REGLAS:
        1. NO uses información de internet.
        2. Si no está en el contrato, responde: "La información solicitada no se encuentra detallada en este contrato."
        3. Sé súper conciso. NUNCA superes las 150 palabras.
        
        [CONTRATO]
        {texto_completo}
        
        [PREGUNTA DEL USUARIO]
        {peticion.pregunta}
        """
        
        respuesta = await modelo.generate_content_async(prompt_qa)
        respuesta_cruda = respuesta.text
        
        # Filtro de caracteres seguros
        texto_limpio = re.sub(r'[^\w\s.,;:!?()\'áéíóúÁÉÍÓÚñÑüÜ-]', '', respuesta_cruda)
        texto_limpio = texto_limpio.replace('\n', ' ').replace('\r', ' ')
        texto_limpio = " ".join(texto_limpio.split()) 
        
        datos = {"respuesta": texto_limpio}
        
        del texto_completo
        del prompt_qa
        
        return datos

    except Exception as e:
        print(f"🔥 ERROR INTERNO EN PREGUNTAS: {str(e)}")
        return {"respuesta": f"Error de procesamiento: {str(e)[:100]}"}


# --- ENDPOINT DE DIAGNÓSTICO (EL ESCÁNER) ---
@app.get("/modelos-disponibles")
async def listar_modelos():
    try:
        modelos_permitidos = []
        for m in genai.list_models():
            if 'generateContent' in m.supported_generation_methods:
                modelos_permitidos.append(m.name)
        return {"modelos_que_puedes_usar": modelos_permitidos}
    except Exception as e:
        return {"error_google": str(e)}

    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
